96 bool ErriezMHZ19B::isWarmingUp()
97 {
98  // Wait at least 3 minutes after power-on
99  if (millis() < MHZ19B_WARMING_UP_TIME_MS) {
100 #ifdef MHZ19B_SMART_WARMING_UP
101  static int16_t _lastCO2 = -1;
102  int16_t co2;
103 
104  // Sensor returns valid data after CPU reset and keep sensor powered
105  co2 = readCO2();
106  if (_lastCO2 == -1) {
107  _lastCO2 = co2;
108  } else {
109  if (_lastCO2 != co2) {
110  // CO2 value changed since last read, no longer warming-up
111  _tLastReadCO2 = 0;
112  return false;
113  }
114  }
115 #endif
116  // Warming-up
117  return true;
118  }
119 
120  // Not warming-up
121  return false;
122 }

256 {

258 
259  // Send command "Read range"
260  // NOT DOCUMENTED!
261  sendCommand(MHZ19B_CMD_GET_RANGE);
262 
263  // Read response
264  if (receiveResponse(_response, sizeof(_response)) == MHZ19_RESULT_OK) {
265  range = (_response[4] << 8) | _response[5];

355 void ErriezMHZ19B::sendCommand(uint8_t cmd, byte b3, byte b4, byte b5, byte b6, byte b7)
356 {
357  uint8_t txBuffer[9] = { 0xFF, 0x01, cmd, b3, b4, b5, b6, b7, 0x00 };

359  // Save command for response verification

378 MHZ19B_Result_e ErriezMHZ19B::receiveResponse(uint8_t *rxBuffer, uint8_t rxBufferLength)
379 {
380  unsigned long tStart;
381  MHZ19B_Result_e result;
382 
383  // Argument check
384  if (rxBufferLength < MHZ19B_RESPONSE_LENGTH) {
385  return MHZ19_RESULT_ARGUMENT_ERROR;
386  }
387 
388  // Clear receive buffer
389  memset(rxBuffer, 0, MHZ19B_RESPONSE_LENGTH);
390 
391  // Wait until all data received from sensor
392  tStart = millis();
393  while (serialAvailable() < rxBufferLength) {
394  if ((millis() - tStart) >= MHZ19B_SERIAL_RX_TIMEOUT_MS) {
395  return MHZ19_RESULT_ERR_TIMEOUT;
396  }
397  }
398 
399  // Read response from serial buffer
400  serialRead(rxBuffer, MHZ19B_RESPONSE_LENGTH);
401 
402  // Set result to OK
403  result = MHZ19_RESULT_OK;
404 
405  // Check received Byte[0] == 0xFF and Byte[1] == transmit command
406  if ((rxBuffer[0] != 0xFF) || (rxBuffer[1] != _cmd)) {
407  result = MHZ19_RESULT_ERROR;
408  }
409 
410  // Check received Byte[8] CRC
411  if (rxBuffer[8] != calcCRC(rxBuffer)) {
412  result = MHZ19_RESULT_ERR_CRC;
413  }
414 
415  // Return result
416  return result;
417 }

430 uint8_t ErriezMHZ19B::calcCRC(uint8_t *data)
431 {
432  byte crc = 0;
433 
434  for (uint8_t i = 1; i < 8; i++) {
435  crc += data[i];

437  crc = 0xFF - crc;
438  crc++;
